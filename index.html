                var Perfect_Check = (n) => 
                {
                  let sum= 0;
                  for(let i=1; i<=n/2; i++){
                    if(n%i === 0){
                      sum = sum+i;
                    }
                  }
                  if(sum === n && sum !== 0){
                    return "YES"
                  }
                  else{
                    return  "NO"
                  }
                };
